-- Find user with most rated movies
WITH UserRatings AS (
    SELECT u.name, COUNT(*) AS num_movies
    FROM Users u
    JOIN MovieRating mr ON u.user_id = mr.user_id
    GROUP BY u.name
),
TopUser AS (
    SELECT name
    FROM UserRatings
    WHERE num_movies = (SELECT MAX(num_movies) FROM UserRatings)
    ORDER BY name
    LIMIT 1
),

-- Find movie with highest average rating in Feb 2020
FebRatings AS (
    SELECT m.title, AVG(mr.rating) AS avg_rating
    FROM Movies m
    JOIN MovieRating mr ON m.movie_id = mr.movie_id
    WHERE mr.created_at >= '2020-02-01' AND mr.created_at <= '2020-02-29'
    GROUP BY m.title
),
TopMovie AS (
    SELECT title
    FROM FebRatings
    WHERE avg_rating = (SELECT MAX(avg_rating) FROM FebRatings)
    ORDER BY title
    LIMIT 1
)

-- Combine results
SELECT name AS results FROM TopUser
UNION ALL
SELECT title AS results FROM TopMovie;
